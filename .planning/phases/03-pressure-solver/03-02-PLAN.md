    ```python
    import pytest
    from brewos.models.inputs import SimulationInput, RoastLevel, Mode

    AEROPRESS_STANDARD = SimulationInput(

        })
        steep_result = immersion_accurate(steep_only_inp)
        hybrid_result = simulate(AEROPRESS_STANDARD)
        assert hybrid_result.extraction_yield > steep_result.extraction_yield + 1.0

    def test_aeropress_ey_range():
        result = simulate(AEROPRESS_STANDARD)
        assert 15.0 <= result.extraction_yield <= 25.0

    def test_aeropress_complete_output():
        result = simulate(AEROPRESS_STANDARD)
        assert result.tds_percent > 0
        assert result.extraction_yield > 0
        assert len(result.extraction_curve) > 0
        assert len(result.psd_curve) > 0
        assert result.flavor_profile is not None
        assert result.brew_ratio > 0
        assert result.mode_used == "accurate"

    def test_aeropress_fast_accuracy():
        acc = simulate(AEROPRESS_STANDARD)
        fast_inp = AEROPRESS_STANDARD.model_copy(update={"mode": Mode.fast})
        fast = simulate(fast_inp)
        assert abs(acc.extraction_yield - fast.extraction_yield) < 2.0

    def test_aeropress_fast_speed():
        fast_inp = AEROPRESS_STANDARD.model_copy(update={"mode": Mode.fast})
        start = time.perf_counter()
        for _ in range(100):
            simulate(fast_inp)
        elapsed = (time.perf_counter() - start) / 100
        assert elapsed < 0.001
    ```

    **Step 2: Implement `methods/aeropress.py`** (GREEN phase):

    AeroPress is a standalone method file (NOT a thin dispatcher). It contains its own solver logic because it composes two phases.

    ```python
    """AeroPress -- standalone hybrid solver (immersion steep + pressure push)."""

    import numpy as np
    from scipy.integrate import solve_ivp

    from brewos.models.inputs import SimulationInput
    from brewos.models.outputs import SimulationOutput, ExtractionPoint
    from brewos.utils.params import derive_immersion_params, kozeny_carman_permeability, K_liang, E_max, rho_w
    from brewos.utils.co2_bloom import co2_bloom_factor
    from brewos.utils.output_helpers import resolve_psd, estimate_flavor_profile, generate_warnings, brew_ratio_recommendation

    AEROPRESS_DEFAULTS = {
        "steep_time_s": 60.0,       # typical AeroPress steep
        "push_time_s": 30.0,        # plunge duration
        "push_pressure_bar": 0.5,   # manual hand pressure
        "bed_depth_m": 0.030,       # AeroPress chamber depth
        "bed_diameter_m": 0.063,    # AeroPress inner diameter
        "porosity": 0.40,
        "method_type": "aeropress",
        "ey_target_pct": 19.0,
    }
    ```

    `_solve_hybrid_accurate(inp)`:
    1. Resolve PSD and get Moroney params via `derive_immersion_params()`
    2. **Steep phase:** Run Moroney 3-ODE (same equations as immersion.py) for `steep_time_s` seconds using `solve_ivp(method='Radau')`. Extract final c_h, c_v, psi_s from solution.
    3. **Push phase:** Using final steep state as ICs, run a pressure-push ODE for `push_time_s` seconds:
       - Compute K_bed via `kozeny_carman_permeability(grind_size_um*1e-6, porosity)`
       - Darcy velocity: `q = K_bed / mu * (push_pressure_bar * 1e5) / bed_depth_m`, capped at 5e-3 m/s
       - `mu = 0.3e-3` (water viscosity at 93C)
       - V_bed_pore = `A_bed * bed_depth_m * porosity`
       - ODE adds advective loss term to dc_h: `dc_h = -kA*(c_h-c_v) + kB*(c_sat-c_h)*psi_s - (q/max(V_bed_pore, 1e-9))*c_h`
       - Other equations same as Moroney
    4. **Combine:** Total EY from push phase final c_h (which already includes steep contribution since ICs carry over)
    5. **Scale** to ey_target_pct using Liang-style scaling
    6. **Build extraction_curve**: Concatenate steep curve + push curve time points
    7. Return SimulationOutput with all 7 fields

    `_solve_hybrid_fast(inp)`:
    1. Resolve PSD, compute R_brew
    2. EY_eq = ey_target_pct (19.0%)
    3. Biexponential with AeroPress-calibrated constants: A1=0.55, tau1=4.0, tau2=60.0 (between immersion and moka -- AeroPress has a push but lower pressure than moka)
    4. `EY_t = EY_eq * (1 - A1*exp(-t/tau1) - A2*exp(-t/tau2))`
    5. Return SimulationOutput with mode_used="fast"

    `simulate(inp)`:
    - Mode.accurate -> `_solve_hybrid_accurate(inp)`
    - Mode.fast -> `_solve_hybrid_fast(inp)`

    **Note:** Do NOT modify `immersion.py`. AeroPress duplicates the Moroney ODE internally for the steep phase to avoid coupling with Phase 1 code. This is intentional per RESEARCH.md recommendation.

    ```python
    """Cross-method smoke tests — all 6 brew methods x 2 modes."""

    import pytest
    from brewos.models.inputs import SimulationInput, RoastLevel, Mode

    from brewos.methods.french_press import simulate as french_press
    from brewos.methods.v60 import simulate as v60
    from brewos.methods.kalita import simulate as kalita
    from brewos.methods.espresso import simulate as espresso
    from brewos.methods.moka_pot import simulate as moka_pot
    from brewos.methods.aeropress import simulate as aeropress

    # Standard scenarios per method type
    _SCENARIOS = {
        "french_press": dict(coffee_dose=15.0, water_amount=250.0, water_temp=93.0, grind_size=700.0, brew_time=240.0, roast_level=RoastLevel.medium),
        "v60":          dict(coffee_dose=15.0, water_amount=250.0, water_temp=93.0, grind_size=600.0, brew_time=180.0, roast_level=RoastLevel.medium),
        "kalita":       dict(coffee_dose=15.0, water_amount=250.0, water_temp=93.0, grind_size=600.0, brew_time=180.0, roast_level=RoastLevel.medium),
        "espresso":     dict(coffee_dose=18.0, water_amount=36.0,  water_temp=93.0, grind_size=200.0, brew_time=25.0,  roast_level=RoastLevel.medium, pressure_bar=9.0),
        "moka_pot":     dict(coffee_dose=15.0, water_amount=150.0, water_temp=93.0, grind_size=400.0, brew_time=180.0, roast_level=RoastLevel.medium),
        "aeropress":    dict(coffee_dose=15.0, water_amount=200.0, water_temp=93.0, grind_size=500.0, brew_time=90.0,  roast_level=RoastLevel.medium),
    }

    _METHODS = {
        "french_press": french_press,
        "v60": v60,
        "kalita": kalita,
        "espresso": espresso,
        "moka_pot": moka_pot,
        "aeropress": aeropress,
    }

    @pytest.mark.parametrize("method_name", _METHODS.keys())
    @pytest.mark.parametrize("mode", [Mode.fast, Mode.accurate])
    def test_method_returns_valid_output(method_name, mode):
        """Every method x mode combination returns a complete SimulationOutput."""
        scenario = dict(_SCENARIOS[method_name])
        scenario["mode"] = mode
        inp = SimulationInput(**scenario)
        result = _METHODS[method_name](inp)

        assert result.tds_percent > 0, f"{method_name}/{mode.value}: tds_percent must be > 0"
        assert result.extraction_yield > 0, f"{method_name}/{mode.value}: extraction_yield must be > 0"
        assert 5.0 <= result.extraction_yield <= 30.0, f"{method_name}/{mode.value}: EY {result.extraction_yield}% outside 5-30% plausible range"
        assert len(result.extraction_curve) > 0, f"{method_name}/{mode.value}: extraction_curve empty"
        assert len(result.psd_curve) > 0, f"{method_name}/{mode.value}: psd_curve empty"
        assert result.flavor_profile is not None, f"{method_name}/{mode.value}: flavor_profile is None"
        assert result.brew_ratio > 0, f"{method_name}/{mode.value}: brew_ratio must be > 0"
        assert result.mode_used in ("fast", "accurate"), f"{method_name}/{mode.value}: invalid mode_used"

    @pytest.mark.parametrize("method_name", _METHODS.keys())
    def test_method_distinct_ey(method_name):
        """Each method returns EY > 0 (not stuck at zero)."""
        scenario = dict(_SCENARIOS[method_name])
        scenario["mode"] = Mode.fast
        inp = SimulationInput(**scenario)
        result = _METHODS[method_name](inp)
        assert result.extraction_yield > 5.0, f"{method_name}: EY suspiciously low ({result.extraction_yield}%)"
    ```
